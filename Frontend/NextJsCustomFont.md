#### we can use custom fonts in next js app like this:

**structure**

```bash
/app
  /fonts
    Texta-Regular.woff2
    Texta-Bold.woff2
/lib
  fonts.ts  <-- fonts config here
/app
  layout.tsx

```

```jsx
// lib/fonts.ts
import localFont from "next/font/local";
import { Lato } from "next/font/google";

// Texta (Local Font)
export const texta = localFont({
  src: [
    {
      path: "../app/fonts/Texta-Regular.woff2",
      weight: "400",
      style: "normal",
    },
    {
      path: "../app/fonts/Texta-Bold.woff2",
      weight: "700",
      style: "normal",
    },
  ],
  variable: "--font-texta",
  display: "swap",
});

// Lato (Google Font)
export const lato = Lato({
  subsets: ["latin"],
  weight: ["400", "700"],
  variable: "--font-lato",
  display: "swap",
});
```

```jsx
// app/layout.tsx
import { texta, lato } from "@/lib/fonts";

export const metadata = {
  title: "Create Next App",
  description: "Generated by create next app",
};

export default function RootLayout({
  children,
}: {
  children: React.ReactNode,
}) {
  return (
    <html lang="en">
      <body className={`${texta.variable} ${lato.variable} antialiased`}>
        {children}
      </body>
    </html>
  );
}
```

```jsx
// tailwind.config.ts
import type { Config } from "tailwindcss";

const config: Config = {
  theme: {
    extend: {
      fontFamily: {
        texta: "var(--font-texta)",
        lato: "var(--font-lato)",
      },
    },
  },
  plugins: [],
};
export default config;
```

**So now you can easily use in your app:**

```jsx

<h1 className="font-texta text-4xl">
  Texta heading
</h1>

<p className="font-lato text-base">
  Lato paragraph
</p>``
```
